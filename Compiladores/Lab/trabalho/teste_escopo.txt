int x = 10;  // variavel global

float soma(int a, int b) {
    int x = 5;      // variavel local com mesmo nome da global
    float result;   // variavel local da função
    
    if (a > 0) {
        int y = 20; // variavel do bloco if
        x = x + y;  // usa o x local da função
    }
    
    result = a + b + x;
    return result;
}

int main() {
    int z = 30;     // variavel local do main
